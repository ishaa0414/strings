           for(let i=0;i< n;i++) {
             let count =0;
             for(let j=0;j< n;j++) {

                 break;

                 count+=1;
                 
               }
               
             }

               ans+=str[i]+count;

Reverse the String

        var Reverse_String = (str) => 
        {
          let newString = "";

            }
